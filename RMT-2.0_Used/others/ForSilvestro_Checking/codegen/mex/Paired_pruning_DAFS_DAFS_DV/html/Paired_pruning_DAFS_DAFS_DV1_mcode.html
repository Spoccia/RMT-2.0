1function [remainQOctave,Dist] = Paired_pruning_DAFS_DAFS_DV(feature1,depdScale1,matches,combineScore,feature2,depdScale2,doctave,toctave )
  2% pruning considering that two feature overlapping in time should have also
  3% same variate
  4    remainQOctave=zeros(doctave,toctave);
  5    Dist=0;
  6    for ii = 1 : doctave
  7        for  jj = 1 : toctave
  8            index1 = find(feature1(5,:) == ii & feature1(6,:) == jj);
  9            matches11 = matches(:, index1);
 10            combineScore11 = combineScore(index1);
 11
 12            [diagY,I] = sort(combineScore11,2,'descend');
 13            diagY_1=diagY;
 14            diagMatch = matches11(:,I);
 15            diagRemainMatch = diagMatch;
 16            counter=0;
 17%             Counteroccurrences= 0
 18       %     i=1;
 19            dimention=size(diagRemainMatch,2);
 20            i=1;
 21            while i<=dimention
 22                
 23                centerc_variate = feature1(1,diagRemainMatch(1,i));
 24                centerm_variate = feature2(1,diagRemainMatch(2,i));
 25            
 26                centerc = feature1(2,diagRemainMatch(1,i));
 27                centerm = feature2(2,diagRemainMatch(2,i));
 28                rangec = 3*(feature1(4,diagRemainMatch(1,i))) ;
 29                rangem = 3*(feature2(4,diagRemainMatch(2,i))) ;
 30                startc = centerc - rangec ;
 31                endc = centerc + rangec ;
 32                startm = centerm - rangem ;
 33                endm = centerm + rangem ;
 34
 35                keept = true;
 36                keepd = true;
 37
 38                depdin1 = depdScale1(:,diagRemainMatch(1,i));
 39                depdin1 = depdin1(depdin1~=0);
 40                depdin2 = depdScale2(:,diagRemainMatch(2,i));
 41                depdin2 = depdin2(depdin2~=0);
 42                over = overlap (depdin1,depdin2);
 43                if(over < 0.5)
 44                    keepd = false;
 45                end
 46
 47                j = 1;
 48                 
 49              
 50                while  (j<=counter) && (keept == true)&& (keepd == true)%(j<=size(diagRemainMatch,2)) && (keept == true)&& (keepd == true)
 51                    % first step check only time
 52                   
 53                        rem_variatec = feature1(1,diagRemainMatch(1,j));
 54                        rem_variatem = feature2(1,diagRemainMatch(2,j));    
 55                    
 56                        rem_centerc = feature1(2,diagRemainMatch(1,j));
 57                        rem_centerm = feature2(2,diagRemainMatch(2,j));
 58                        rem_rangec = 3*(feature1(4,diagRemainMatch(1,j))) ;
 59                        rem_rangem = 3*(feature2(4,diagRemainMatch(2,j))) ;
 60                        rem_startc = rem_centerc - rem_rangec ;
 61                        rem_endc = rem_centerc + rem_rangec ;
 62                        rem_startm = rem_centerm - rem_rangem ;
 63                        rem_endm = rem_centerm + rem_rangem ;
 64%                         
 65%                         VariateF1 = feature1(1,diagRemainMatch(1,j));
 66%                         VariateF2 = feature2(1,diagRemainMatch(2,j));
 67                        list1 = zeros(4,1);%[];
 68                        list2 = zeros(4,1);%[];
 69
 70
 71                        list1(4-3,1) =rem_startc;% [list1;rem_startc;rem_endc];
 72                        list1(4-2,1) = rem_endc;
 73                        list2(4-3,1) = rem_startm;%[list2;rem_startm;rem_endm];
 74                        list2(4-2,1) = rem_endm;
 75                        list1(4-1,1) = startc;% [list1;startc;endc];
 76                        list1(4,1)   = endc;%
 77                        list2 (4-1,1) = startm;%[list2;startm;endm];
 78                        list2 (4,1) = endm;%
 79
 80                        [c d] = sort(list1);
 81                        rankcstart = find(c==startc);
 82                        rankcend = find(c==endc);
 83                        [c d] = sort(list2);
 84                        rankmstart = find(c==startm);
 85                        rankmend = find(c==endm);
 86
 87                        if (startc == rem_startc) && (endc==rem_endc) && (startm == rem_startm) && (endm == rem_endm)                            
 88                            if(centerc_variate == rem_variatec)
 89                                keept = false;                            
 90                                break;
 91                            else
 92                                keept = true;
 93                            end
 94                        else
 95                            if rankcstart(1) == rankmstart(1) && rankcend(1) ==rankmend(1)
 96                                keept = true;
 97                            else
 98                                keept = false;
 99                                break;
100                            end
101                        end
102               
103                    j = j+1;
104                end
105                if (keept == true) && (keepd == true)
106%                     diagRemainMatch(i) =1;% [diagRemainMatch diagMatch(:,i)];
107                    Dist = Dist + diagY_1(i);
108                    counter=counter+1;
109                else
110                    diagRemainMatch(:,i)=[];
111                    diagY_1(:,i)=[];
112                    i=i-1;
113                    dimention=size(diagRemainMatch,2);
114                end
115                i=i+1;
116            end
117            % diagRemainFeature = feature1(:,diagRemainMatch(1,:));
118            TT = counter;% size(diagRemainMatch, 2);
119            remainQOctave(ii,jj) = TT;%[remainQOctave TT];
120        end
121    end
